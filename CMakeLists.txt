cmake_minimum_required(VERSION 3.10)
project(megalo VERSION 0.1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(PkgConfig REQUIRED)
pkg_check_modules(LV2 REQUIRED lv2)

# ── Options ───────────────────────────────────────────────────────────────
option(MEGALO_PHASE_VOCODER
    "Enable phase vocoder pitch shifter (higher quality, more CPU)" OFF)
set(MEGALO_PV_N "2048" CACHE STRING
    "Phase vocoder FFT size — must be a power of 2 (1024 for MOD Dwarf, 2048 for Pi 5)")

option(MEGALO_RAVE
    "Enable RAVE neural synthesis decoder stub (RPi 5 only — requires trained model + ONNX RT)" OFF)
set(MEGALO_HN_QUALITY "2" CACHE STRING
    "Polyphonic analysis tier: 0=dwarf (A35), 1=pi5 (A76), 2=desktop")

# Build switches for the two distinct plugins (both default ON).
option(MEGALO_BUILD_STD "Build the stock granular Megalo plugin" ON)
option(MEGALO_BUILD_HN  "Build the polyphonic harmonic+noise MegaloHN plugin" ON)

set(MEGALO_SOURCES src/plugin.cpp src/megalo_dsp.cpp)
# Unified factory presets (generated by tools/gen_presets.py — keep in sync).
set(MEGALO_PRESETS
    Clean_Sustain.ttl Octave_Pad.ttl Shimmer.ttl Choir.ttl Cathedral.ttl
    Drone.ttl Glass.ttl Sub_Bass.ttl Octaver.ttl Frozen_Strings.ttl)

# Shared per-target setup: compile flags, the optional phase-vocoder defines,
# the LV2 output name/bundle, and the TTL + modgui copy step.
#   _target  CMake target name
#   _name    .so output name (without prefix)
#   _bundle  source/build bundle directory name (e.g. megalo.lv2)
function(megalo_configure_target _target _name _bundle)
    set(_bundle_dir "${CMAKE_BINARY_DIR}/${_bundle}")

    target_include_directories(${_target} PRIVATE ${LV2_INCLUDE_DIRS} src)
    target_compile_options(${_target} PRIVATE
        -Wall -Wextra -Wno-unused-parameter
        -ffast-math
        -fvisibility=hidden
        $<$<CONFIG:Release>:-O3>
        $<$<CONFIG:Debug>:-O0 -g>
    )

    if(MEGALO_PHASE_VOCODER)
        target_compile_definitions(${_target} PRIVATE
            MEGALO_PHASE_VOCODER MEGALO_PV_N=${MEGALO_PV_N})
    endif()

    set_target_properties(${_target} PROPERTIES
        PREFIX ""
        OUTPUT_NAME "${_name}"
        LIBRARY_OUTPUT_DIRECTORY "${_bundle_dir}"
    )

    # Copy the manifest + plugin TTL + modgui TTL + presets into the bundle.
    set(_ttls
        "${CMAKE_SOURCE_DIR}/${_bundle}/manifest.ttl"
        "${CMAKE_SOURCE_DIR}/${_bundle}/${_name}.ttl"
        "${CMAKE_SOURCE_DIR}/${_bundle}/modgui.ttl")
    foreach(_preset ${MEGALO_PRESETS})
        list(APPEND _ttls "${CMAKE_SOURCE_DIR}/${_bundle}/${_preset}")
    endforeach()

    add_custom_command(TARGET ${_target} POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy ${_ttls} "${_bundle_dir}/"
        COMMAND ${CMAKE_COMMAND} -E copy_directory
            "${CMAKE_SOURCE_DIR}/${_bundle}/modgui" "${_bundle_dir}/modgui"
        COMMENT "Copying TTL + modgui files to ${_bundle}")

    install(DIRECTORY "${_bundle_dir}" DESTINATION lib/lv2)
endfunction()

# ── Stock granular Megalo (MEGALO_HN_SYNTH off) ────────────────────────────
if(MEGALO_BUILD_STD)
    add_library(megalo MODULE ${MEGALO_SOURCES})
    megalo_configure_target(megalo "megalo" "megalo.lv2")
    message(STATUS "Megalo (granular) enabled")
endif()

# ── Polyphonic MegaloHN (MEGALO_HN_SYNTH on) ───────────────────────────────
if(MEGALO_BUILD_HN)
    add_library(megaloHN MODULE ${MEGALO_SOURCES})
    megalo_configure_target(megaloHN "megaloHN" "megaloHN.lv2")
    target_compile_definitions(megaloHN PRIVATE
        MEGALO_HN_SYNTH MEGALO_HN_QUALITY=${MEGALO_HN_QUALITY})
    message(STATUS "MegaloHN (H+N additive, tier ${MEGALO_HN_QUALITY}) enabled")
    if(MEGALO_RAVE)
        target_compile_definitions(megaloHN PRIVATE MEGALO_RAVE)
        message(STATUS "RAVE stub enabled (RPi 5 — model + ONNX RT required)")
        # TODO: find_package(ONNXRuntime REQUIRED)
        # target_link_libraries(megaloHN PRIVATE onnxruntime)
    endif()
endif()
